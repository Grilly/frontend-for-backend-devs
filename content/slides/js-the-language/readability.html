Readability (1/7)

Use string interpolation instead of concatenation with template literalsES2015: 


        // concatenation
        const a = 1;
        const b = 'a equals ' + a + '.';

        b;  // 'a equals 1.'
      

        // interpolation
        const a = 1;
        const b = `a equals ${a}.`;

        b;  // 'a equals 1.'

Readability (2/7)

In objects…

  • when key and value have the same name, use property shorthandES2015: 
    
          const a = 1, b = 2;
          const o = { a: a, b: b };
        
    
          const a = 1, b = 2;
          const o = { a, b };
  • use method shorthandES2015: 
    
          const o = {
            a: function(b, c) {}
          };
        
    
          const o = {
            a(b, c) {}
          };
  • when needed, use computed property namesES2015: 
    
          const a = 'computed';
          const o = {};
          o[a] = 1;
          
    
          const a = 'computed';
          const o = { [a]: 1 };

Readability (3/7)

  • Functions can have default parametersES2015: 
    
          const a = (b) => {
            b = b || 1;
            return b;
          }

          a();  // 1
        
    
          const a = (b = 1) => b




          a();  // 1
  • ⚠️ It only works with undefined: 
    
        const a = (b = 1) => b

        a(0);         // 0
        a(null);      // null
        a(undefined); // 1

Readability (4/7)

  • It's possible to destructureES2015 properties from objects: 
    
          const o = { a: 1, nested: { b: 2 } };
          const a = o.a;
          const b = o.nested.b;
          a;  // 1
          b;  // 2
        
    
          const o = { a: 1, nested: { b: 2 } };
          const { a, nested: { b } } = o;

          a;  // 1
          b;  // 2
  • With default values: 
    
            const o = { a: 1 };
            const { a, b = 2 } = o;
            a;  // 1
            b;  // 2
    With arrays: 
    
            const arr = [1, 2];
            const [a, b] = arr;
            a;  // 1
            b;  // 2
  • With function parameters (and default values): 
    
        const fn = ({ a, b, c = 3 }) = ({ a, b, c });
        fn({ a: 1, b: 2 }); // { a: 1, b: 2, c: 3 }

Readability (5/7)

  • With rest parametersES2015 any number of arguments are stored in an array: 
    
        const a = (...args) => args;
        a();      // []
        a(1, 2);  // [1, 2]
  • With the spread operatorES2015 any iterableES2015 (arrays, strings and more) can be "spread" into its distinct values: 
    
          // copy array
          const a = [1, 2, 3];
          const b = [...a];
          b;        // [1, 2, 3]
          a === b;  // false
        
    
          // concat arrays
          const a = [2, 3];
          const b = [1, ...a, 4];
          b;  // [1, 2, 3, 4]

Readability (6/7)

The spread operator also works with objectsES2018: 


        // copy object
        const o1 = {
          a: 1,
          b: { c: 3 },
        };
        const o2 = { ...o1 };

        o2;     // {
                //    a: 1,
                //    b: { c: 3 }
                // }
        o1 === o2;      // false

        // no deep copy:
        o1.c === o2.c;  // true
      

        // merge objects
        const o1 = {
          a: 1,
          b: 2,
        };
        const o2 = {
          ...o1,
          a: 3,
          c: 4,
        };
        o2; // {
            //    a: 3,
            //    b: 2,
            //    c: 4,
            // }

Readability (7/7)

  • Prefer declarative Array.prototype methods over imperative loops: 
    
          const a = [1, 2, 3];
          let b = [];
          for (const i = 0; i < a.length; i++) {
            b[i] = a[i] * 2;
          }
          b;  // [2, 4, 6]
        
    
          const a = [1, 2, 3];
          const b = a.map((item) => item * 2);



          b;  // [2, 4, 6]
  • More examples: 
    
          const a = [1, 2, 3];
          // filter items:
          const b = a.filter((item) => item > 2);
          b;  // [3];
        
    
          const a = [1, 2, 3];
          // find item:
          const b = a.find((item) => item === 2);
          b;  // 2;
        
    
          const a = [1, 2, 3];
          // check if all items satisfy a condition:
          const b = a.every((item) => item > 2);
          b;  // false;
        
    
          const a = [1, 2, 3];
          // reduce to a single value:
          const b = a.reduce((acc, item) => acc + item, 0);
          b;  // 6;
