RxJS

Common operators

Creation (1/5)


      import { from } from 'rxjs';

      from([1, 2, 3]).subscribe((val) => {
        console.log(val);
      });

      // 1
      // 2
      // 3
    

      import { from } from 'rxjs';

      from('abc').subscribe((val) => {
        console.log(val);
      });

      // 'a'
      // 'b'
      // 'c'
    

    import { from } from 'rxjs';

    from(Promise.resolve('a')).subscribe((val) => {
      console.log(val);
    });

    // 'a'

Transformation (2/5)


      import { of } from 'rxjs';
      import { pluck } from 'rxjs/operators';

      of({ a: 1 }, { a: 2 })
        .pipe(
          pluck('a')
        )
        .subscribe((val) => {
          console.log(val);
        });

      // 1
      // 2
    

      import { of } from 'rxjs';
      import { mergeMap } from 'rxjs/operators';

      of(1, 2)
        .pipe(
          mergeMap((val) => of(val * 2))
        )
        .subscribe((val) => {
          console.log(val);
        });

      // 2
      // 4
    

    import { of } from 'rxjs';
    import { switchMap } from 'rxjs/operators';

    of(1, 2)
      .pipe(
        switchMap((val) => of(val * 2))
      )
      .subscribe((val) => {
        console.log(val);
      });

    // 4
